95  static UINT CalcConstantBufferByteSize(UINT byteSize)
96  {
97  // Constant buffers must be a multiple of the minimum hardware
98  // allocation size (usually 256 bytes). So round up to nearest
99  // multiple of 256. We do this by adding 255 and then masking off
100  // the lower 2 bytes which store all bits < 256.
101  // Example: Suppose byteSize = 300.
102  // (300 + 255) & ~255
103  // 555 & ~255
104  // 0x022B & ~0x00ff
105  // 0x022B & 0xff00
106  // 0x0200
107  // 512
108  return (byteSize + 255) & ~255;
109  }
